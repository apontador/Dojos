A corrente do protótipo

    var Person = function () {
        var greeting = "Hello ";

        this.name = null;

        this.sayName = function () {
            console.log(greeting + this.name);
        }
    }

    Person.prototype.setName(name) {
        this.name = name;
    }

A corrente do protótipo

    var Developer = function () {
        this.surname = null;
    }

    Developer.prototype = new Person();
    Developer.prototype.setSurname = function(surname) {
        this.surname = surname;
    }

    var developer = new Developer();

    console.log(developer.__proto__) // Person

        lorem = "ipsun";

    function fighter () {
        var foo = "fighter";

        console.log("foo " + foo);

    }

    fighter() // foo fighter

    console.log("lorem " + lorem);  // lorem ipsun
